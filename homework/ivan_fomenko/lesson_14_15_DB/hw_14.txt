SELECT 
	s.name AS name,
	s.second_name AS last_name,
	g.title AS group_name,
	b.title AS took_book,
	l.id  AS lesson_id,
	l.title AS lessons,
	m.value AS marks
FROM students AS s 
JOIN `groups` AS g 
	ON s.group_id = g.id 
LEFT JOIN books AS b
	ON b.taken_by_student_id = s.id
LEFT JOIN marks AS m
	ON m.student_id = s.id
LEFT JOIN lessons AS l
	ON l.id = m.lesson_id 
WHERE s.id = 22202;
